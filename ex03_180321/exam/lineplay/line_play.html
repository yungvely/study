(function (win, $, doc) {
    var lineplay = {
        init: function() {
            this.setElements();
            this.setArray(); 
            this.setIndex();                   
            this.bindEvents();         
        },
        setElements: function() {
            this.obj = $('.swiper-wrapper');
            this.objList = this.obj.find('.swiper-slide');
            this.btn = $('button');
        },
        bindEvents : function() {
            this.btn.on('click', $.proxy(this.onClick, this));
        },
        onClick : function (e) {
            // var target = $(e.currentTarget);
            e.preventDefault();
            this.viewFunc();
        },
        setArray: function () {
            this.array = [];
            for (var i = 0; i < this.objList.length; i++) {
                this.array.push(i);
            }
            // console.log('처음',this.array);
        },
        resetArray : function (idx) {
            this.array.splice(idx, 1);
            if (this.array.length) return;
            Math.floor(Math.random() * this.array.length);
        },
        setIndex : function () {
            var random = Math.floor(Math.random() * this.array.length),
                currentIndex = this.array[random];
            this.objList.eq(currentIndex).show().siblings().hide();
            this.oldNum = currentIndex;
            this.resetArray(random);
            // console.log('랜덤',this.array);
        },
        viewFunc: function() {
            var random = Math.floor(Math.random() * this.array.length),
                currentIndex = this.array[random];
            if (this.array.length == 0) {
                this.setArray(); 
                this.setIndex();
            }
            this.objList.eq(currentIndex).show().siblings().hide();
            // console.log('이전지워진',this.oldNum,'지워질',currentIndex,this.array);
            this.resetArray(random);
            this.oldNum = currentIndex;
        }
    }
    lineplay.init();
})(window, window.jQuery, window.document);
