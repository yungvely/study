    (function (win, $) {
        // checkbox
        var checkbox = function (container, args) {
            var defParams = {
                container : container || '.box',
                chkBoxWrap : '.check_box',
                inputChkBox : '.ipt_chk',
                checkActive : 'chk_active'
            }
            this.opts = $.extend(defParams, (args || {}));
            this.init();
        }
        checkbox.prototype = {
            init : function () {
                this.setElements();
                this.bindEvents();
                // 체크된 상태면 ((active 클레스가 있으면 실제 checked 넣어주기))
                $('.'+this.opts.checkActive+' '+this.opts.inputChkBox).prop('checked',true);
            },
            setElements : function () {
                this.inputChkBox = $(this.opts.container).find(this.opts.inputChkBox); 
            },        
            bindEvents : function () {
                this.inputChkBox.on('change', $.proxy(this.onChangeProps, this));
            },
            onChangeProps : function () {                
                if (this.inputChkBox.prop('checked')) {
                    $(this.opts.chkBoxWrap).addClass(this.opts.checkActive);
                } else {
                    $(this.opts.chkBoxWrap).removeClass(this.opts.checkActive);
                }
            },
        };
        // radio
        var radio = function (container, args) {
            var defParams = {
                container : container || '.box',
                radioBoxWrap : '.radio_box',
                inputRadio : '.ipt_rdo',
                checkActive : 'chk_active'
            }
            this.opts = $.extend(defParams, (args || {}));
            this.init();
        }
        radio.prototype = {
            init : function () {
                this.setElements();
                this.bindEvents();
            },
            setElements : function () {
                this.radioBoxWrap =$(this.opts.radioBoxWrap);
                this.inputRadio = $(this.opts.inputRadio); 
            },        
            bindEvents : function () {
                this.radioBoxWrap.on('change', '> input', $.proxy(this.onChangeProps, this));
            },
            onChangeProps : function (e) { 
                var classTarget =$(e.delegateTarget);
                this.radioBoxWrap.removeClass(this.opts.checkActive);         
                classTarget.addClass(this.opts.checkActive); 
            },
        };

        // selectbox
        var selectbox = function (container, args) {
            var defParams = {
                container : container || '.box',
                selectedMenu : '.select_menu',
                openClass : 'opened'
            }
            this.opts = $.extend(defParams, (args || {}));
            this.init();
        }
        selectbox.prototype = {
            init : function () {
                this.setElements();
                this.bindEvents();
                // 처음에 닫기 
                this.closeFunc();
            },
            setElements : function () {
                this.selectBox = $(this.opts.container);
                this.selectlistWrap = this.selectBox.find('ul');
                this.selectlist = this.selectlistWrap.find('li');
                this.selectedMenu = this.selectBox.find(this.opts.selectedMenu);
            },        
            bindEvents : function () {
                this.selectBox.on('click', $.proxy(this.openFunc, this));
                this.selectBox.on('clickoutside', $.proxy(this.closeFunc, this));
                this.selectlist.on('click', '>a' , $.proxy(this.selectListFunc, this))
            },
            openFunc : function () { 
                console.log(this.selectBox);
                this.selectBox.toggleClass(this.opts.openClass);
                this.selectlistWrap.toggle();
                // 열릴때 선택이라는 글자를 보여주도록 임의로 설정
                this.selectedMenu.text('선택');
            },
            selectListFunc : function (e) {
                e.preventDefault();
                e.stopPropagation();
                var target = $(e.currentTarget),
                    selectText = target.text();
                this.selectedMenu.text(selectText);
                this.closeFunc();
            },
            closeFunc : function () { 
                this.selectBox.removeClass(this.opts.openClass);
                this.selectlistWrap.hide();
            }
        };

        $.fn.pluginCall = function () {
            for (var i = 0, max = this.length; i < max; i++) {
                new selectbox(this.eq(i),);
            }
        };

        $(function () {
            var checkboxCall = new checkbox('.box1');
            var radioCall = new radio('.box2');

            // select_box로 제어 
            // var selectCall = new selectbox('.select_box');

            $('.select_box').pluginCall();
        });
    })(window, window.jQuery);
