                // javascript code ((이윤호 실장님 코드 -> 자바스크립트가 엄청 깔끔해보인다))
                window.onload = function() {
                    var examObj1 = document.getElementsByClassName('exam_wrap')[0];
                    examObj1.getElementsByTagName('button')[0].onclick = function() {
                        var examChild = examObj1.getElementsByClassName('exam_q')[0].querySelectorAll('*');
                        for (var i = 0; i < examChild.length; i++) {
                            examChild[i].style.backgroundColor = 'yellow';
                        }
                    };
                    var examObj = document.getElementsByClassName('exam_wrap');
                    for (var obj = 1; obj < examObj.length; obj++) {
                        examObj[obj].getElementsByTagName('button')[0].onclick = function() {
                            var examChild = this.parentNode.getElementsByClassName('exam_q')[0];
                            switch (this.parentNode) {
                                case examObj[1]:
                                    examTarget(examChild.querySelectorAll('span'));
                                    break;
                                case examObj[2]:
                                    examTarget(examChild.querySelectorAll('em span'));
                                    break;
                                case examObj[3]:
                                    examTarget(examChild.querySelectorAll('.exam_q > span'));
                                    break;
                                case examObj[4]:
                                    examTarget(examChild.querySelectorAll('div + p'));
                                    break;
                                case examObj[5]:
                                    examTarget(examChild.querySelectorAll('div ~ span'));
                                    break;
                                case examObj[6]:
                                    examTarget(examChild.querySelectorAll('.active'));
                                    break;
                                case examObj[7]:
                                    examTarget(examChild.querySelectorAll('#active'));
                                    break;
                            };
                        };
                    };

                    function examTarget(examNode) {
                        for (var i = 0; i < examNode.length; i++) {
                            examNode[i].style.backgroundColor = 'yellow';
                        }
                    };
                };
