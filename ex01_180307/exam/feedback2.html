                // javascript code ((이윤호 실장님 코드 -> 자바스크립트가 엄청 깔끔해보인다))
                window.onload = function() {
                    var examObj = document.getElementsByClassName('exam_wrap');
                    for (var obj = 0; obj < examObj.length; obj++) {
                        examObj[obj].getElementsByTagName('button')[0].onclick = function() {
                            var examChild = this.parentNode.getElementsByClassName('exam_q')[0];
                            switch (this.parentNode) {
                                case examObj[0]:
                                    examTarget(examChild.querySelectorAll('.exam_q > :first-child'));
                                    break;
                                case examObj[1]:
                                    examTarget(examChild.querySelectorAll('.exam_q > :last-child'));
                                    break;
                                case examObj[2]:
                                    examTarget(examChild.querySelectorAll(':first-child'));
                                    break;
                                case examObj[3]:
                                    examTarget(examChild.querySelectorAll(':last-child'));
                                    break;
                                case examObj[4]:
                                    examTarget(examChild.querySelectorAll(':nth-child(2)'));
                                    break;
                                case examObj[5]:
                                    examTarget(examChild.querySelectorAll(':nth-child(3)'));
                                    break;
                                case examObj[6]:
                                    examTarget(examChild.querySelectorAll(':nth-child(3) ~ *'));
                                    break;
                                case examObj[7]:
                                    examTarget(examChild.querySelectorAll(':nth-child(1), :nth-child(2)'));
                                    break;
                            };
                        };
                    };

                    function examTarget(examNode) {
                        for (var i = 0; i < examNode.length; i++) {
                            examNode[i].style.backgroundColor = 'yellow';
                        }
                    };
                };

                $(function() {
                    $('.exam_wrap').on('click', 'button', function() {
                        var examCont = $(this).siblings('.exam_q');
                        var idx = $(this).closest('.exam_wrap').index();
                        switch (idx) {
                            case 0:
                                examCont.find('div:first').css('background', 'yellow');
                                break;
                            case 1:
                                examCont.find('div:last').css('background', 'yellow');
                                break;
                            case 2:
                                examCont.find('div').children().first().css('background', 'yellow');
                                break;
                            case 3:
                                examCont.find('div').children().last().css('background', 'yellow');
                                break;
                            case 4:
                                examCont.find('div:nth-child(2)').children().css('background', 'yellow');
                                break;
                            case 5:
                                examCont.find('div:nth-child(3)').css('background', 'yellow');
                                break;
                            case 6:
                                examCont.find('div:nth-child(4)').nextAll().css('background', 'yellow');
                                break;
                            case 7:
                                examCont.find('div:nth-child(3)').prevAll().css('background', 'yellow');
                                break;
                        }
                    });
                });
