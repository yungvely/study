Q. for/while 문 2개로 구구단을 출력하세요.


var exam4 = $('.exam_q').eq(3);
	// Q4 - 한단을 돌떄 아홉번의 루프가 추가적으로 돌아야 함.
    var targetArray = [];

    for (var i = 2; i < 10; i++) {
        // 배열을 사용해 prototype중 하나를 사용해 보자.
        targetArray.push('<'+'div class="inline-block">');
        targetArray.push('<'+'h3>'+i + '단'+'<'+'/h3>');
        for (var j = 1; j < 10; j++) {
            targetArray.push('<'+'span>');
            targetArray.push(i + ' X ' + j + ' = ' + (i * j));

            targetArray.push('<'+'/span><'+'br/>');
        }
        targetArray.push('<'+'/div>');
    }

    exam4.get(0).innerHTML = targetArray.join('');
    // get -> 태그 돔 형태로 바꿔주는 역활을 하므로 써준다.
    // join -> 콤마를 지우거나 바꿔주기 위해서
